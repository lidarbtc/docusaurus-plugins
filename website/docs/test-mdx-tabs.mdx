#!/bin/bash

# Function to calculate factorial
factorial() {
  local n=$1
  if [ $n -le 1 ]; then
    echo 1
  else
    local prev=$(factorial $((n-1)))
    echo $((n * prev))
  fi
}

result=$(factorial 5)
echo "Factorial of 5 is: $result"
